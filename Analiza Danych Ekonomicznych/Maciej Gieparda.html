def popraw_nazwy_kolumn(df):
    # Zamiana nazw kolumn na małe litery
    df.rename(columns=lambda x: x.lower(), inplace=True)
    
    # Zamiana polskich znaków na łacińskie litery
    try:
        df.columns = [col.encode('latin1').decode('unicode_escape').encode('latin1').decode('ascii', 'ignore') for col in df.columns] 
    except:   
        df.columns = [col.translate(str.maketrans("ąćęłńóśźż", "acelnoszz")) for col in df.columns]
    
    try:
        df.drop(columns=['brana'], inplace=True)
    except:
        pass

    # Poprawianie specyficznych nazw kolumn
    df.rename(columns={
        'ceni': 'plec',
        'wykszta': 'wyksztalcenie',
        'roczny_dochd': 'roczny_dochod',
        'wydatki_ywno': 'wydatki_zywnosc',
        'osoby_zalene': 'osoby_zalezne',
        'oszczdnoci': 'oszczednosci',
        'inwestycje_nieruchomoci': 'inwestycje_nieruchomosci',
        'lata_dowiadczenia': 'lata_doswiadczenia'
    }, inplace=True)



    return df

def oblicz_udzial_w_dochodzie(df):
    wydatki_kolumny = ['wydatki_mieszkanie', 'wydatki_zywnosc', 'wydatki_transport', 'wydatki_edukacja']
    for kolumna in wydatki_kolumny:
        nowa_kolumna = f'{kolumna}_udzial'
        df[nowa_kolumna] = (df[kolumna] / df['roczny_dochod'])
    
    return df
